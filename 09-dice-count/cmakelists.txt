cmake_minimum_required(VERSION 3.10)
project(diceCount)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Find OpenCV package
find_package(OpenCV REQUIRED)

# Add the executable
add_executable(diceCount diceCount.cpp)

# Link OpenCV libraries
target_link_libraries(diceCount PRIVATE ${OpenCV_LIBS})
